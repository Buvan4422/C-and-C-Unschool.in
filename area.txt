#include<iostream>
#include<math.h>
using namespace std;
int main()
{
  float  a, b, c, s, radius, area, l, w;
 int ch;
  cout<<"1.Area of circle";
  cout<<"\n2.Area of triangle" ;
  cout<<"\n3.Area of rectangle" ;
  cout<<"\n  Enter your choice :- " ;

   cin>>ch ;

   switch(ch)
   {
    case 1:
  cout << "Enter radius of circle: ";
  cin >> radius;

  area = 3.14*radius*radius;
  cout << "Area = " << area << endl;
  break ;

   case 2:
       {
        cout << "Enter lengths of sides of triangle: ";
        cin >> a>>b>>c;
        s=(a+b+c)/2;
        area = sqrt(s*(s-a)*(s-b)*(s-c));
       cout<< "Area of triangle = " << area<<endl;
       break ;
       }
  case 3:
      {
         cout<< "\nEnter lengths of sides of rectangle: ";
         cin>> l>>w;
          area = l*w ;
         cout << "Area of rectangle = " <<area<<endl;
        break ;
      }
  default : cout<< "\nInvalid Input ";
  break ;
    return 0;
   }

}
